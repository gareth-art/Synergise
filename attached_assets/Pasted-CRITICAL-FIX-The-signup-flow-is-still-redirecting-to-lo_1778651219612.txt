## STEP 2: THE ROOT CAUSE IS ALMOST CERTAINLY ONE OF THESE

After diagnosing, fix whichever of these is causing the problem:

### Likely Cause A: ProtectedRoute redirecting authenticated users away from /signup
If the /signup route is wrapped in any auth guard that says "if authenticated, redirect to /dashboard or /login" — this fires immediately after signup succeeds and overrides the setLocation('/onboarding') call.

Fix: The /signup and /login routes must be completely unwrapped — no ProtectedRoute, no AuthGuard, no redirect logic of any kind. They are public routes.

```tsx
// In App.tsx — signup and login must be plain routes, no wrapper:
<Route path="/signup" component={Signup} />
<Route path="/login" component={Login} />
// NOT:
// <ProtectedRoute path="/signup" ... /> ← wrong
```

### Likely Cause B: A global useEffect in App.tsx redirecting after auth state updates
If App.tsx has a useEffect like:
```tsx
useEffect(() => {
  if (user && !onboardingComplete) setLocation('/login');
  // or
  if (user) setLocation('/dashboard');
}, [user]);
```
This fires whenever auth state changes — including right after signup — and overrides the signup page's own navigation.

Fix: Remove any global redirect useEffect from App.tsx entirely. Redirects must only happen in ProtectedRoute components and explicitly in page-level handlers, never in App.tsx.

### Likely Cause C: The onboarding guard redirecting to /login on 401
If the ProtectedRoute checks onboarding status and gets a 401 (because session isn't set yet) and redirects to /login — this creates the loop.

Fix: ProtectedRoute must show a loading spinner on ANY non-200 response, not redirect to /login. Only redirect to /login when the auth check explicitly returns 401 with isLoading === false.

### Likely Cause D: signup.tsx still has an onAuthSuccess callback
If signup.tsx calls an onAuthSuccess prop or callback after signup that is defined elsewhere and redirects to /login.

Fix: Remove onAuthSuccess entirely from signup.tsx. Post-signup navigation must be self-contained in signup.tsx only.

### RequireAuth component:
```tsx
function RequireAuth({ children }: { children: React.ReactNode }) {
  const { data: user, isLoading } = useQuery({
    queryKey: ['/api/auth/me'],
    queryFn: async () => {
      const res = await fetch('/api/auth/me', { credentials: 'include' });
      if (res.status === 401) return null;
      if (!res.ok) throw new Error('Auth check failed');
      return res.json();
    },
    staleTime: 60 * 1000,
    retry: false
  });

  if (isLoading) return <LoadingSpinner message="Loading..." />;
  if (!user) {
    setLocation('/login');
    return null;
  }
  return <>{children}</>;
}
```

### RequireOnboarding component:
```tsx
function RequireOnboarding({ children }: { children: React.ReactNode }) {
  const { data: profile, isLoading } = useQuery({
    queryKey: ['/api/onboarding'],
    queryFn: async () => {
      const res = await fetch('/api/onboarding', { credentials: 'include' });
      if (res.status === 404) return null;
      if (!res.ok) return null; // treat errors as "not done" only for 404
      return res.json();
    },
    staleTime: 5 * 60 * 1000,
    retry: false
  });

  if (isLoading) return <LoadingSpinner message="Setting up your dashboard..." />;
  if (!profile) {
    setLocation('/onboarding');
    return null;
  }
  return <>{children}</>;
}
```

### Signup.tsx — post-submit handler (definitive version):
```tsx
const onSubmit = async (data: FormData) => {
  setIsLoading(true);
  setError(null);
  try {
    const res = await fetch('/api/auth/signup', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      credentials: 'include',
      body: JSON.stringify({
        fullName: data.fullName,
        email: data.email,
        password: data.password,
        username: data.email
      })
    });
    const result = await res.json();
    if (!res.ok) {
      setError(result.error || 'Signup failed. Please try again.');
      return;
    }
    // Set auth state in cache
    queryClient.setQueryData(['/api/auth/me'], result.user);
    // Clear any stale onboarding cache
    queryClient.removeQueries({ queryKey: ['/api/onboarding'] });
    // Navigate to onboarding
    setLocation('/onboarding');
  } catch {
    setError('Connection error. Please try again.');
  } finally {
    setIsLoading(false);
  }
};
```

### Login.tsx — post-submit handler (definitive version):
```tsx
const onSubmit = async (data: FormData) => {
  setIsLoading(true);
  setError(null);
  try {
    const res = await fetch('/api/auth/login', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      credentials: 'include',
      body: JSON.stringify({ email: data.email, password: data.password })
    });
    const result = await res.json();
    if (!res.ok) {
      setError('Incorrect email or password');
      return;
    }
    // Set auth state
    queryClient.setQueryData(['/api/auth/me'], result.user);
    // Check onboarding
    const onboardingRes = await fetch('/api/onboarding', { credentials: 'include' });
    if (onboardingRes.ok) {
      setLocation('/dashboard');
    } else {
      setLocation('/onboarding');
    }
  } catch {
    setError('Connection error. Please try again.');
  } finally {
    setIsLoading(false);
  }
};
```
